\begin{frame}{RSA decryption/signing}
\begin{itemize}
\item RSA common method to implement digital signatures or public-key encryption
\item math: $c^d \pmod n$ where $d$ secret value
\item how to compute $c^d \pmod n$? one idea:
    \begin{itemize}
    \item say $d=2^0 + 2^3 + 2^8$
    \item then $m^d = m^{2^0 + 2^3 + 2^4} = m\cdot m^{2^3}\cdot m^{2^8} = m\cdot(((m^2)^2)^2)\cdot((((m^2)^2)^2))^2$
    \item trick: compute $m$, $m\cdot m = m^2 \pmod n$, $m^2\cdotm^2 = m^4 \pmod n$, \ldots
    \item and conditionally multiply with result
    \end{itemize}
\item overall algorithm:
    \begin{itemize}
    \item for each bit of d:
        \begin{itemize}
        \item compute $c^{2^i}$ mod $n$ using $c^{2^{i-1}} \cdot c^{2^{i-1}}$
        \item if bit set in $d$, multiply with result and take mod $n$
        \end{itemize}
    \end{itemize}
\end{itemize}
\end{frame}
