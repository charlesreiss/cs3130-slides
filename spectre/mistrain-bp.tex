\begin{frame}[fragile]{mistraining branch predictor?}
\begin{lstlisting}[language=C]
if (something) {
    CodeToRunSpeculatively()
}
\end{lstlisting}
\begin{itemize}
\item useful for attacks: \\
   have `something' be false, but predicted
    as true
\vspace{.5cm}
\item one way: run lots of times with something true then do actually run with something false
\item another way: learn how branch prediction caches work, run code that fills in caches in known way
\end{itemize}
\end{frame}
