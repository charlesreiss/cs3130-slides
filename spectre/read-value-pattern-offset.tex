\begin{frame}[fragile]{reading a value --- * X * BLOCK\_SIZE}
\begin{lstlisting}[language=C,style=smaller]
char *array;
posix_memalign(&array, CACHE_SIZE, CACHE_SIZE);
AccessAllOf(array);
other_array[mystery * 8 * BLOCK_SIZE] += 1;
for (int i = 0; i < CACHE_SIZE; i += BLOCK_SIZE) {
    if (CheckIfSlowToAccess(&array[i])) {
        ...
    }
}
\end{lstlisting}
\begin{itemize}
\item if other\_array is char* and starts at multiple of cache size
\item and \texttt{array[0x100*BLOCK\_SIZE]} is slow to access
\item mystery * 8 = 0x100 + K * SET\_COUNT
\item mystery = 32 + K * SET\_COUNT / 8
\end{itemize}
\end{frame}
