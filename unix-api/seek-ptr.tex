

\begin{frame}[fragile]{unshared seek pointers}
if ``foo.txt'' contains ``AB''
\begin{Verbatim}[fontsize=\small]
int fd1 = open("foo.txt", O_RDONLY);
int fd2 = open("foo.txt", O_RDONLY);
char c;
read(fd, &c, 1);
char d;
read(fd, &d, 1);
\end{Verbatim}
expected result: c = `A', d = `A'
\end{frame}
