\begin{frame}[fragile]{a bad idea}
\begin{itemize}
\item one \myemph{bad} idea to wait for an event:
\end{itemize}
\begin{lstlisting}
bool ready = false;
void WaitForReady() {
    do {} while (!ready);
}

void MarkReady() {
    ready = true;
}
\end{lstlisting}
\begin{itemize}
\item wastes processor time
\item and also \myemph{doesn't work!}
\end{itemize}
\end{frame}
