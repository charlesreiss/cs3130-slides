\begin{lstlisting}[language=C++,style=smaller]
pthread_rwlock_t lock;
void ThreadA() {
  pthread_rwlock_rdlock(&lock);
  puts("a");
  ...
  puts("A");
  pthread_rwlock_unlock(&lock);
}
void ThreadB() {
  pthread_rwlock_rdlock(&lock);
  puts("b");
  ...
  puts("B");
  pthread_rwlock_unlock(&lock);
}
\end{lstlisting}
\end{minipage}

  pthread_rwlock_wrlock(&lock);
  puts("d");
