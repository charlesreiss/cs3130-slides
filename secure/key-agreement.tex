
\begin{frame}
\frametitle{key agreement}
    \begin{itemize}
    \item problem: A has B's public encryption key \\
        wants to choose shared secret
    \vspace{.5cm}
    \item some ideas:
        \begin{itemize}
        \item A chooses a key, sends it encrypted to B
        \item A sends a public key encrypted to B, B chooses a key and sends back
        \end{itemize}
    \item TLS, SSH use special key agreement primitive instead
        \begin{itemize}
        \item easier to use temporary keys to limit effect of key compromise
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Diffie-Hellman-style key agreement}
    \begin{itemize}
    \item public/private-key-like:
        \begin{itemize}
        \item  ``key share'' (public) 
        \item private (random) value
        \end{itemize}
    \item key share = GenerateKeyShare(private value)
    \item KeyGen(A's private value, B's key share) = new shared secret
    \item KeyGen(B's private value, A's key share) = same shared secret
    \vspace{.5cm}
    \item math providers gaurentee:
        \begin{itemize}
        \item hard to derive shared secret without one of the private values
        \end{itemize}
    \end{itemize}
\end{frame}
